	  //랜덤 발생 0~255

	  c = parseInt( Math.random()*255); 
	  
	  console.log(a, b, c);
	  
	  pobj.style.backgroundColor = 'rgba('  + 
			    a + ',' +  b + ',' + c + ')';
 }
